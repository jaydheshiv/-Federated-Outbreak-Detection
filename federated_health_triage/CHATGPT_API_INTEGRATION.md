```python
import openai
from typing import Optional
import os

class ChatGPTClinicalAdvisor:
    """Integrate ChatGPT for clinical explanations"""
    
    def __init__(self, api_key: Optional[str] = None):
        """Initialize ChatGPT client"""
        self.api_key = api_key or os.getenv('OPENAI_API_KEY')
        
        if not self.api_key:
            raise ValueError(
                "OpenAI API key not found. "
                "Set OPENAI_API_KEY environment variable or "
                "pass api_key parameter."
            )
        
        openai.api_key = self.api_key
        self.model = "gpt-4"  # or "gpt-3.5-turbo" for faster/cheaper
    
    def explain_risk_level(self, 
                          patient_data: dict,
                          risk_level: int,
                          confidence: float) -> str:
        """Generate clinical explanation for predicted risk"""
        
        prompt = f"""
        You are a clinical epidemiologist. Provide a brief clinical explanation
        for this patient's infection risk assessment.
        
        PATIENT DATA:
        - Age: {patient_data.get('age', 'Unknown')}
        - Symptoms: {patient_data.get('symptoms', [])}
        - Vaccination Status: {patient_data.get('vaccination_status', 'Unknown')}
        - Proximity to Confirmed Cases: {patient_data.get('proximity_to_confirmed', 0)}
        - Travel History: {patient_data.get('travel_history', 0)}
        - Pre-existing Conditions: {patient_data.get('comorbidities', 'None')}
        
        PREDICTION RESULTS:
        - Predicted Risk Level: {risk_level} (0=Low, 1=Moderate, 2=High, 3=Critical)
        - Model Confidence: {confidence*100:.1f}%
        
        Please provide:
        1. Summary assessment (2-3 sentences)
        2. Top 3 risk factors contributing to this risk level
        3. Recommended immediate actions (testing, isolation, etc.)
        4. Follow-up timeline
        
        Keep response concise and clinically appropriate.
        """
        
        response = openai.ChatCompletion.create(
            model=self.model,
            messages=[
                {
                    "role": "system",
                    "content": "You are a clinical epidemiologist providing brief, actionable clinical assessments."
                },
                {
                    "role": "user",
                    "content": prompt
                }
            ],
            temperature=0.3,  # Lower = more deterministic
            max_tokens=500
        )
        
        return response['choices'][0]['message']['content']
    
    def generate_outbreak_report(self,
                                clinic_name: str,
                                num_cases: int,
                                avg_age: float,
                                symptoms_profile: dict,
                                vaccination_gap: float,
                                recommended_actions: list) -> str:
        """Generate professional outbreak report"""
        
        prompt = f"""
        Generate a professional public health outbreak report based on this data:
        
        OUTBREAK DETAILS:
        - Location: {clinic_name}
        - Number of Confirmed Cases: {num_cases}
        - Average Age of Cases: {avg_age}
        - Timeline: Past 7 days
        
        CLINICAL PROFILE:
        - Most Common Symptoms: {', '.join([f"{k}={v}" for k,v in symptoms_profile.items()])}
        - Vaccination Gap: {vaccination_gap*100:.1f}% unvaccinated
        
        ALERT LEVEL: {'HIGH' if num_cases >= 10 else 'MODERATE'}
        
        Please generate:
        1. Executive summary
        2. Epidemiological assessment
        3. Public health recommendations
        4. Monitoring plan for next 7 days
        
        Format as a professional medical report.
        """
        
        response = openai.ChatCompletion.create(
            model=self.model,
            messages=[
                {
                    "role": "system",
                    "content": "You are a public health epidemiologist preparing official outbreak reports."
                },
                {
                    "role": "user", 
                    "content": prompt
                }
            ],
            temperature=0.2,
            max_tokens=1000
        )
        
        return response['choices'][0]['message']['content']
    
    def extract_symptoms_from_text(self, patient_description: str) -> dict:
        """Extract structured symptom data from natural language"""
        
        prompt = f"""
        Extract medical information from this patient description.
        
        PATIENT DESCRIPTION:
        "{patient_description}"
        
        Response as JSON format:
        {{
            "symptoms": [list of symptoms],
            "symptom_count": number,
            "severity": "mild/moderate/severe",
            "duration_days": number,
            "fever_present": true/false,
            "respiratory_symptoms": true/false
        }}
        """
        
        response = openai.ChatCompletion.create(
            model=self.model,
            messages=[
                {
                    "role": "system",
                    "content": "Extract medical information accurately. Response must be valid JSON."
                },
                {
                    "role": "user",
                    "content": prompt
                }
            ],
            temperature=0.1,  # Very deterministic for data extraction
            max_tokens=300
        )
        
        import json
        response_text = response['choices'][0]['message']['content']
        
        try:
            return json.loads(response_text)
        except json.JSONDecodeError:
            # Fallback if JSON parsing fails
            return {"raw_response": response_text}
```

### OpenAI API Pricing (as of 2024)

```
GPT-3.5-Turbo:
├─ Input: $0.0005 per 1K tokens
├─ Output: $0.0015 per 1K tokens
└─ Much faster, lower cost

GPT-4:
├─ Input: $0.03 per 1K tokens
├─ Output: $0.06 per 1K tokens
└─ More accurate, higher cost

COST ESTIMATION for this project:
Running explain_risk_level() for 1000 patients:
├─ GPT-3.5: ~$2-3
├─ GPT-4: ~$15-20
└─ Monthly: $50-100 at scale

Token estimation:
- Prompt: ~200 tokens
- Response: ~300 tokens
- Total: ~500 tokens per call
```

```
GPT-3.5-Turbo:
├─ Response time: 2-5 seconds
└─ Good enough for reports (run in background)

GPT-4:
├─ Response time: 5-10 seconds
└─ Better quality, need async processing

Random Forest (Local ML):
├─ Response time: <0.1 seconds
└─ Instant predictions
```

### Option 3: Cached Responses (Cost Optimization)

```python
import json
from typing import Dict

class CachedChatGPTAdvisor:
    """Cache ChatGPT responses to reduce API calls"""
    
    def __init__(self):
        self.advisor = ChatGPTClinicalAdvisor()
        self.cache_file = "chatgpt_cache.json"
        self.cache = self._load_cache()
    
    def _load_cache(self) -> Dict:
        try:
            with open(self.cache_file, 'r') as f:
                return json.load(f)
        except FileNotFoundError:
            return {}
    
    def _save_cache(self):
        with open(self.cache_file, 'w') as f:
            json.dump(self.cache, f)
    
    def explain_risk_level(self, patient_data, risk_level, confidence):
        """Check cache before calling API"""
        
        # Create cache key from patient characteristics
        cache_key = f"risk_{risk_level}_age_{patient_data.get('age')}_{int(confidence*100)}"
        
        if cache_key in self.cache:
            print(f"✅ Using cached response for {cache_key}")
            return self.cache[cache_key]
        
        # Call API if not cached
        response = self.advisor.explain_risk_level(patient_data, risk_level, confidence)
        
        # Store in cache
        self.cache[cache_key] = response
        self._save_cache()
        
        return response
```
